### **Backend Tracking Functions**

#### **Achievement Tracking** (`achievementService.js`)

```javascript
/**
 * Track achievement progress
 * @param {string} heroId - Hero document ID
 * @param {string} trackingKey - Achievement tracking key (e.g., 'taunts_used')
 * @param {number} increment - Amount to increment by
 */
async function trackAchievement(heroId, trackingKey, increment = 1) {
  // Similar to quest tracking
  const heroRef = db.collection('heroes').doc(heroId);
  
  // Get all achievements that use this tracking key
  const achievements = await db.collection('achievements')
    .where('trackingKey', '==', trackingKey)
    .get();
  
  // Update progress for each
  const updates = {};
  achievements.forEach(doc => {
    const achievement = doc.data();
    const progressPath = `achievements.progress.${doc.id}.current`;
    updates[progressPath] = admin.firestore.FieldValue.increment(increment);
  });
  
  await heroRef.update(updates);
  
  // Check for completion
  await checkAchievementCompletion(heroId, trackingKey);
}

/**
 * Check if any achievements were just completed
 */
async function checkAchievementCompletion(heroId, trackingKey) {
  const heroDoc = await db.collection('heroes').doc(heroId).get();
  const hero = heroDoc.data();
  
  const achievements = await db.collection('achievements')
    .where('trackingKey', '==', trackingKey)
    .get();
  
  achievements.forEach(async (doc) => {
    const achievement = doc.data();
    const progress = hero.achievements?.progress?.[doc.id];
    
    if (progress && !progress.completed && progress.current >= achievement.requirement.count) {
      // Achievement completed!
      await awardAchievement(heroId, doc.id, achievement);
    }
  });
}

/**
 * Award achievement rewards
 */
async function awardAchievement(heroId, achievementId, achievement) {
  const heroRef = db.collection('heroes').doc(heroId);
  
  await heroRef.update({
    [`achievements.completed`]: admin.firestore.FieldValue.arrayUnion(achievementId),
    [`achievements.progress.${achievementId}.completed`]: true,
    [`achievements.progress.${achievementId}.completedAt`]: admin.firestore.Timestamp.now(),
    
    // Unlock rewards
    [`titles.unlocked`]: admin.firestore.FieldValue.arrayUnion(achievement.rewards.title),
    [`badges.unlocked`]: admin.firestore.FieldValue.arrayUnion(achievement.rewards.badge),
    
    // Grant currency
    gold: admin.firestore.FieldValue.increment(achievement.rewards.gold),
    tokens: admin.firestore.FieldValue.increment(achievement.rewards.tokens)
  });
  
  // Broadcast achievement unlocked event
  broadcastToRoom(hero.twitchUserId, {
    type: 'achievement_unlocked',
    achievement: {
      id: achievementId,
      name: achievement.name,
      description: achievement.description,
      icon: achievement.icon,
      rewards: achievement.rewards
    }
  });
}
```
